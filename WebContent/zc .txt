<%
// ============================================
// 数据库配置
// ============================================
String driverName = "com.mysql.jdbc.Driver";
String userName = "root";
String userPasswd = "123456";
String dbName = "bishe";
String tableName = "denglu";
String url = "jdbc:mysql://localhost:3306/" + dbName;
Connection con = null;
Statement st = null;
ResultSet rs = null;

// 加载驱动
try {
	Class.forName(driverName).newInstance();
} catch(ClassNotFoundException e) {
	System.out.print("Error loading Driver,不能加载驱动程序");
	return;
}

// 连接数据库
try {
	con = DriverManager.getConnection(url, userName, userPasswd);
} catch(SQLException er) {
	System.out.print("Error getConnection,不能连接数据库");
	out.print("<script>alert('数据库连接失败');history.back();</script>");
	return;
}

// ============================================
// 注册业务逻辑
// ============================================
request.setCharacterEncoding("UTF-8");   // 保证接收的中文不乱码

// 前端传过来的参数名还是 txtUser（不用改）
String name = request.getParameter("txtUser");
String password = request.getParameter("txtPassword");
String password1 = request.getParameter("txtPassword1");

// 1. 检查是否为空
if(name == null || name.trim().equals("") || 
   password == null || password.trim().equals("")) {
	out.print("<script>alert('用户名和密码不能为空！');history.back();</script>");
	return;
}
// 2. 检查两次密码是否一致
if(!password.equals(password1)) {
	out.print("<script>alert('两次输入的密码不一致！');history.back();</script>");
	return;
}

try {
	st = con.createStatement();
	
	// 3. 检查用户名是否已被注册（字段改为 name）
	String checkSql = "SELECT * FROM " + tableName + " WHERE name='" + name + "'";
	rs = st.executeQuery(checkSql);
	if(rs.next()) {
		out.print("<script>alert('该用户名已被注册，请更换！');history.back();</script>");
		return;
	}
	
	// 4. 插入数据（字段改为 name）
	String sql = "INSERT INTO " + tableName + "(name, password) VALUES (?, ?)";
	PreparedStatement pstmt = con.prepareStatement(sql);
	pstmt.setString(1, name);
	pstmt.setString(2, password);
	
	int result = pstmt.executeUpdate();
	if(result > 0) {
		out.print("<script>alert('注册成功！即将跳转到登录页');window.location.href='login.html';</script>");
	} else {
		out.print("<script>alert('注册失败，请重试');history.back();</script>");
	}
	pstmt.close();
	
} catch(SQLException er) {
	System.err.println("Error executeInsert,不能添加数据！");
	out.print("<script>alert('注册失败：" + er.getMessage() + "');history.back();</script>");
} finally {
	// 关闭资源
	if(rs != null) try { rs.close(); } catch(Exception e) {}
	if(st != null) try { st.close(); } catch(Exception e) {}
	if(con != null) try { con.close(); } catch(Exception e) {}
}
%>
